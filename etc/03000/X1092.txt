<h3 id="X1092">X(1092)&nbsp;=&nbsp;TRILINEAR CUBE OF X(3)</h3>
Trilinears&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; f(A,B,C)
: f(B,C,A) : f(C,A,B),
<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;where
f(A,B,C) = cos<sup>3</sup>A
<br>
Barycentrics g(A,B,C) : g(B,C,A) : g(C,A,B), where g(A,B,C)

</p>
